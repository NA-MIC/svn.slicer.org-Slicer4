PROJECT(SlicerIO)

IF (TCL_LIBRARY)
ELSE(TCL_LIBRARY)

TARGET_LINK_LIBRARIES (slicerio ${TCL_LIBRARY})

TARGET_LINK_LIBRARIES(testSlicerIO slicerio ${TCL_LIBRARY})
